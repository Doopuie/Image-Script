#!/bin/bash

disk=""
path=""
file=""

lsblk

while true; do

	read -p "Enter the USB disk (example: /dev/sdc):" disk

	if [ ! -b disk ]; then
		echo "Invalid disk."
		continue
	fi

	break
done

read -p "WARNING: This will erase everything on $disk. Type YES to continue: " confirm
if [ $confirm != "YES" ]; then
	echo "Aborted."
	exit 1
fi

sudo umount {$disk}* 2>/dev/null

while true; do

	read -p "Enter the folder path containing the .bin file: " path	

	if [ -e path ]; then
		cd $path || exit 1
		break
	else
		echo "Enter a valid path."
	fi
done

cd $path

while true; do
	if [ -f file ]; then
		echo "creating image..."
		break
	else
		read -p "Enter a valid .bin file:" file
	fi
done

while true; do

	read -p "Enter the .bin file name: " file

	if [ -f $file ]; then

		echo "creating image..."
		break
	else
		echo "Enter a valid .bin."
done

sudo dd if=file of=$disk bs=4M status=progress conv=fsync
sync
echo "Your recovery USB is ready."
